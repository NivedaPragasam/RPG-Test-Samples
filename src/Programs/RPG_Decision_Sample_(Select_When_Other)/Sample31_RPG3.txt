// Language: RPGLE (free format) - User action processing with indicators for screen keys
// Map function key indicators to actions via SELECT
dcl-s done ind inz(*off);

exfmt MAIN;  // sets indicators like *in03(F3), *in05(F5), *in12(Cancel)

select;
  when *in03;
     done = *on;                 // F3=Exit
  when *in05;
     refreshData();              // F5=Refresh
  when *in12;
     cancelChanges();            // F12=Cancel
  other;
     processDefault();
endsl;
