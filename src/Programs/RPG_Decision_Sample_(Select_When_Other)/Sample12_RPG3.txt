// Language: RPG (mixed free + fixed) - Legacy fixed inside free-form block for compatibility
// Mixed example: Free-form calls into a legacy fixed-format SELECT group
dcl-s stateCode char(2);
dcl-s region char(10);

stateCode = %trim(stateCode);

// Legacy fixed-format snippet retained
 /free
  // Fall through to fixed-format logic marker
 /end-free
C                   SELECT                   stateCode
C                   WHENEQ                   'CA'
C                   EVAL      region = 'WEST'
C                   WHENEQ                   'NY'
C                   EVAL      region = 'EAST'
C                   WHENEQ                   'TX'
C                   EVAL      region = 'SOUTH'
C                   OTHER
C                   EVAL      region = 'UNKNOWN'
C                   ENDSL
 /free
  if region = 'UNKNOWN';
     logWarning('Unhandled state: ' + stateCode);
  endif;
 /end-free
