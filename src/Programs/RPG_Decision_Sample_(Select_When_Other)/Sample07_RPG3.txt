// Language: RPGIII (fixed format) - Error mapping with OTHER default and subroutine calls
     * Map API error codes to actions
C                   SELECT                   ApiErr
C                   WHENEQ                   0
C                   EXSR      ContinueProc
C                   WHENEQ                   101
C                   EXSR      RetryOnce
C                   WHENEQ                   404
C                   EXSR      NotFound
C                   WHENLT                   0
C                   EXSR      LogSysErr
C                   OTHER
C                   EXSR      LogAppErr
C                   ENDSL
