// Language: RPGLE (free format) - Banking transaction fee rules combining category and amount
// Apply fees by transaction category and amount
dcl-s cat char(10);
dcl-s amt packed(9:2);
dcl-s fee packed(7:2);

select;
  when cat = 'WIRE' and amt <= 1000;
     fee = 15.00;
  when cat = 'WIRE' and amt > 1000;
     fee = 25.00;
  when cat = 'ATM' and amt in %range(0:300);
     fee = 2.50;
  other;
     fee = 0.00;
endsl;
