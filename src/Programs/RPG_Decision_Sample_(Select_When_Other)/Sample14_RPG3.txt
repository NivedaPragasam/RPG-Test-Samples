// Language: RPGIII (fixed format) - Status consolidation using SELECT operand for character codes
     * Map order status code to bucket
C                   SELECT                   OrdStat                * 'N','P','S','C','X'
C                   WHENEQ                   'N'
C                   EVAL      StatBucket='NEW'
C                   WHENEQ                   'P'
C                   EVAL      StatBucket='PICK'
C                   WHENEQ                   'S'
C                   EVAL      StatBucket='SHIP'
C                   WHENEQ                   'C'
C                   EVAL      StatBucket='COMP'
C                   OTHER
C                   EVAL      StatBucket='EXCEPT'
C                   ENDSL
