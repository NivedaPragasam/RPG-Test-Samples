// Language: RPGLE (free format) - Payment method branching including default OTHER
// Route by payment method
dcl-s payMethod char(10);

select;
  when payMethod in %list('CARD':'CREDITCARD');
     processCard();
  when payMethod = 'ACH';
     processACH();
  when payMethod = 'CASH';
     processCash();
  other;
     logError('Unsupported payment: ' + payMethod);
endsl;
