// Language: RPGLE (free format) - API response handling combining numeric and character checks
// Interpret REST status and message
dcl-s httpCode int(10);
dcl-s msg char(50);

select;
  when httpCode in %range(200:299);
     msg = 'OK';
  when httpCode = 401 or httpCode = 403;
     msg = 'AUTH REQUIRED';
  when httpCode = 404;
     msg = 'NOT FOUND';
  when httpCode in %range(500:599);
     msg = 'SERVER ERROR';
  other;
     msg = 'UNKNOWN STATUS';
endsl;
