// Language: RPGLE (free format) - Escalation rules with nested SELECT and indicators
// Ticket priority vs SLA breach
dcl-s priority char(1); // 'L','M','H'
dcl-s slaBreached ind;

select;
  when priority = 'H';
     escalateImmediate();
  when priority = 'M';
     select;
       when slaBreached;
         escalateToTier2();
       other;
         assignToQueue('M-TIER1');
     endsl;
  when priority = 'L';
     assignToQueue('L-POOL');
  other;
     assignToQueue('UNKNOWN');
endsl;
