// Language: RPGIII (fixed format) - Menu option processing with SELECT operand
     * Main menu option: 1=Orders,2=Customers,3=Reports,9=Exit
C                   SELECT                   MenuOpt
C                   WHENEQ                   1
C                   EXSR      GoOrders
C                   WHENEQ                   2
C                   EXSR      GoCustomers
C                   WHENEQ                   3
C                   EXSR      GoReports
C                   WHENEQ                   9
C                   EVAL      *INLR = *ON
C                   OTHER
C                   EXSR      BadOption
C                   ENDSL
