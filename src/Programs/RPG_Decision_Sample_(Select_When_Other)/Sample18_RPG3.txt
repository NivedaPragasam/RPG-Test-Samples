// Language: RPGIII (fixed format) - Batch posting state machine using SELECT operand
     * Batch posting phases
C                   SELECT                   Phase                  * 'LOAD','VALID','POST','DONE'
C                   WHENEQ                   'LOAD'
C                   EXSR      LoadData
C                   WHENEQ                   'VALID'
C                   EXSR      ValidateData
C                   WHENEQ                   'POST'
C                   EXSR      PostData
C                   WHENEQ                   'DONE'
C                   EXSR      Cleanup
C                   OTHER
C                   EXSR      FailState
C                   ENDSL
