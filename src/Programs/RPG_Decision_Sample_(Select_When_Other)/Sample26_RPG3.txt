// Language: RPGIII (fixed format) - Log routing by severity and subroutines
     * Severity: 'D','I','W','E','F'
C                   SELECT                   Sev
C                   WHENEQ                   'D'
C                   EXSR      LogDebug
C                   WHENEQ                   'I'
C                   EXSR      LogInfo
C                   WHENEQ                   'W'
C                   EXSR      LogWarn
C                   WHENEQ                   'E'
C                   EXSR      LogError
C                   WHENEQ                   'F'
C                   EXSR      LogFatal
C                   OTHER
C                   EXSR      LogUnknown
C                   ENDSL
