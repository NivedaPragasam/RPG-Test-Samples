// Language: RPGLE (free format) - IN using %RANGE and sparse %LIST combined
// Validate code is either in a list or in a numeric range
dcl-s code int(10);

select;
  when code in %range(100:199);          // family A
     route('FAM-A');
  when code in %list(210:211:212:299);   // sparse hits
     route('FAM-B');
  other;
     route('OTHER');
endsl;
