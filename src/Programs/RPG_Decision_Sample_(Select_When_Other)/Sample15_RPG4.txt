// Language: RPGLE (free format) - Holiday checks using %lookup on date array
// Check if date is in list of holidays
dcl-s workDate date;
dcl-ds holidays dim(5) qualified;
  d date;
end-ds;

holidays(1).d = %date('2025-01-01');
holidays(2).d = %date('2025-07-04');
holidays(3).d = %date('2025-11-27');
holidays(4).d = %date('2025-12-25');
holidays(5).d = %date('2025-12-31');

select;
  when %lookup(workDate: holidays(*).d) > 0;
     markHoliday();
  other;
     normalWorkDay();
endsl;
