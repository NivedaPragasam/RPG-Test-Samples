// Language: RPGLE (free format) - Pointer null checks (WHEN equals/NOT equals semantics)
// Treat pointer states as NULL/NOT NULL checks
dcl-s custPtr pointer;
dcl-s msg char(30);

select;
  when custPtr = *null;              // NULL pointer
     msg = 'No customer loaded';
  when custPtr <> *null;             // NOT NULL pointer
     msg = 'Customer loaded';
  other;
     msg = 'Unknown pointer state';
endsl;
