// Language: RPGLE (free format) - “IN” list using %LIST for multiple SKU flags
// Combine logic for multiple special SKUs
dcl-s sku char(10);
dcl-s label char(20);

select;
  when sku in %list('GIFT-10':'PROMO-99':'SAMPLE-01');
     label = 'NON-STOCK';
  when %lookup(sku:%list('HAZ-001':'HAZ-002':'HAZ-003')) > 0;
     label = 'HAZMAT';
  other;
     label = 'STANDARD';
endsl;
