// Language: RPGLE (free format) - Parsing action tokens with %UPPER, SELECT without operand
// Token-based action selection
dcl-s token char(8);

token = %upper(%trim(token));

select;
  when token = 'SYNC';
     doSync();
  when token = 'RESET';
     doReset();
  when token in %list('ON':'OFF');
     togglePower(token);
  other;
     logError('Bad token: ' + token);
endsl;
